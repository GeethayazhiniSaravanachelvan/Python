s=input("enter the password")
a=('!','@','#','$','%','^','&','*','(',')','-','_','=','+')
l=len(s)
if l<8:
   print("password invalid")
else:
  u=l=b=r=m=0
  for ele in s:
      if ele.isupper():
       u=True
      if ele.islower():
       l=True
      if ele.isdigit():
       m=True
      if ele in a:
       r=True
  if u and l  and r and m==True:
       print("valid password")
  else:
       print("invalid password")
